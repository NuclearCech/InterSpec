# Step 1) Check out InterSpec and all of its dependencies (besides boost and Wt).  
#         (You can skip this step if you have already checked out the code.)
git clone --recursive https://github.com/sandialabs/interspec/

# Step 2) Cd to the directory with the Docker files
cd interspec/target/docker/

# Step 3) build and install boost 1.65.1 and Wt 3.3.4 into an Ubuntu 18.0.4 base image
docker build -t interspec_ubuntu_build_env:v1.0 -f "setup_build_env_ubuntu.dockerfile" ../..

# Step 4) Compile InterSpec.
docker build -t interspec_ubuntu_build:v1.0 -f "build_interspec_ubuntu.dockerfile" ../..

# Step 5) Enter docker image with compiled InterSpec.  The below command will let you use gdb to debug if you want and it maps the images port 8080 to your machines port 8080.
docker run --cap-add=SYS_PTRACE --security-opt seccomp=unconfined -p 8080:8080 -i -t -v /Users/wcjohns/rad_ana/InterSpec:/interspec interspec_ubuntu_build:v1.0 bash

# Step 6) From within docker, run InterSpec
cd /tmp/build_interspec/
./bin/InterSpec.exe --docroot . --http-address 0.0.0.0 --http-port 8080 -c ./data/config/wt_config_localweb.xml

# Step 7) Point your browser to http://localhost:8080/ and use InterSpec.
